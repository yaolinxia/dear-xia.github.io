import time

name = input("请输入您的姓名：")
sex = input("请输入您的性别：")
phone = input("请输入您的联系方式：")

#模拟打印的过程
print("系统正在打印中...1")
time.sleep(1)
print("系统正在打印中...2")
time.sleep(1)
print("系统正在打印中...3")
time.sleep(1)


#2.获取用户信息并打印
print("========================================")
print("姓名：" + name)
print("性别：" + sex)
print("联系方式：" + phone)
